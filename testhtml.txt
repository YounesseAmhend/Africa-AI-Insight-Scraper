      $(function() {
        function update(by) {
          var parameters = {};
          var year = $('.content-area #year-button').data('value');
          var month = $('.content-area #month-button').data('value');
          var category = $('.content-area #category-button').data('value');
          var tag = $('.content-area .tags a.active').data('value');
          var byDate = false;
          if (year && month) {
            parameters.year = year;
            parameters.month = month;
            byDate = true;
          }
          if (category) {
            parameters.category = category;
          }
          if (tag) {
            parameters.tag = tag;
          }
          if (by === "category-button" || by === "tag" || byDate) {
            if (!$.isEmptyObject(parameters)) {
              window.location = window.location.pathname+"?"+$.param(parameters);
            } else {
              window.location = window.location.pathname;
            }
          }
        }
        $('.content-area #year-dropdown a, .content-area #month-dropdown a, .content-area #category-dropdown a').click(function(){
          var button = $(this).parents('.btn-group').find('button:first');
          if (button.data('value') !== $(this).data('value')) {
            button.data('value', $(this).data('value')).text($(this).text());
            update(button.attr('id'));
          }
        });
        $('.content-area .tags a').click(function(e){
          e.preventDefault();
          alreadyActive = $(this).hasClass('active');
          $('.content-area .tags a').removeClass('active');
          if (!alreadyActive) {
            $(this).addClass('active');
          }
          update('tag');
        });
      });
